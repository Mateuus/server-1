// ---------- Lanzador de Mob (Snake) -------------------------------
// ------------------------------------------------------------------
payon,1,1,1	script	CazeriaPayon01	66,{

OnInit:
	disablenpc "CazeriaPayon01";
	end;

OnEnable:
monster "payon",0,0,"Snake",1025,85,"CazeriaPayon01::OnMyMobDead";
end;

OnMyMobDead:
	set $mobcazeria,$mobcazeria-1;
	set @puntoscazeria,@puntoscazeria+1;
	dispbottom "Recibes 1 Punto de Cacería";
	dispbottom "Tienes un total de "+@puntoscazeria+" Punto/s de Cacería";
	if ($mobcazeria < 1)goto fincaza;
end;

fincaza:
	mapannounce "payon","Organizador Cacería: ¡Fin de la Cacería!",bc_yellow;
	sleep2 2000;
	mapannounce "payon","Organizador Cacería: ¡Teneis 30 minutos para venir a recoger vuestros premios!",bc_yellow;
	sleep2 3000;
	mapannounce "payon","Organizador Cacería: ¡Podreis encontrarme cerca del Tool Dealer, al sur de Payon!",bc_yellow;
	set $CazeriaPayon,2;
	sleep2 1800000;
	mapannounce "payon","Organizador Cacería: ¡El tiempo para cambiar vuestros puntos ha terminado!",bc_yellow;
	set $CazeriaPayon,0;
	sleep2 2000;
	mapannounce "payon","Organizador Cacería: ¡Espero ver vuestras caras de nuevo en la próxima cacería!",bc_yellow;
	goto inicio;
end;}

-	script	GMNAmes	-1,{
     
	OnPCLoginEvent:
	if(getgroupid() >= 50 ) end;
	set .Yourname$,strcharinfo(0);
	setarray .GmNames$[0],"GM","Admin","Staff";
	for( set .x, 0; .x < getarraysize(.GmNames$); set .x, .x +1)
	if(compare(.Yourname$,.GmNames$[.x])) {
	mes "¿Estás tratando de ser falso GM?";
	mes "Usted expulsado automático del servidor";
	sleep2 3000;
	atcommand "@block "+strcharinfo(0);
	end;
}
	percentheal 100,100;
	end;
}
